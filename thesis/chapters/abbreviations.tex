\chapter*{List of Abbreviations and Symbols}
\addcontentsline{toc}{chapter}{List of Abbreviations and Symbols}

\begin{description}
    \item[AI] Artificial Intelligence
    \item[AUC] Area Under the Curve
    \item[AUPRC] Area Under the Precision-Recall Curve
    \item[AURC] Area Under the Risk-Coverage Curve
    \item[ECE] Expected Calibration Error
    \item[H\&E] Hematoxylin and Eosin
    \item[MC Dropout] Monte Carlo Dropout
    \item[MSP] Maximum Softmax Probability
    \item[NLL] Negative Log-Likelihood
    \item[OOD] Out-of-Distribution
    \item[PCAM] PatchCamelyon
    \item[PR AUC] Area Under the Precision-Recall Curve
    \item[ROC AUC] Area Under the Receiver Operating Characteristic Curve
    \item[ViT] Vision Transformer
    \item[WSI] Whole-Slide Image
    \item[$u(x)$] Uncertainty score assigned to sample $x$
    \item[$p(y \mid x)$] Predictive probability distribution over classes for input $x$
    \item[$H[p(y \mid x)]$] Predictive entropy of the class distribution
\end{description}
